
SELECT	session_id as [Session ID], 
		[Command], 
		a.text AS Query, 
		start_time [Inicio em], 
		percent_complete [Conclusão em %], 
		Status,
		blocking_session_id [ID Sessão Block],
		dateadd(second,estimated_completion_time/1000, getdate()) as [Estimativa de Conclusão]
FROM sys.dm_exec_requests r CROSS APPLY sys.dm_exec_sql_text(r.sql_handle) a
--WHERE r.command in ('BACKUP DATABASE','RESTORE DATABASE')
